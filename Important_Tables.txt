        "Month Number", MONTH([Date]),
        "Weekday", FORMAT([Date],"ddd"),
        "Weekday Number", WEEKDAY([Date]),
        "Weekday Type", IF(WEEKDAY([Date])=1, "Weekend", IF(WEEKDAY([Date])=7, "Weekend", "Weekday")),
        "Qtr", "Qtr " & QUARTER([Date])
    )
